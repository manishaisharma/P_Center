SELECT count(*) FROM   REP_USERS,        REP_VERSION_PROPS,        REP_SUBJECT ,opb_object_type WHERE  REP_USERS.USER_ID = REP_VERSION_PROPS.USER_ID               AND REP_SUBJECT.SUBJECT_ID = REP_VERSION_PROPS.SUBJECT_ID        and SUBJECT_AREA='masharma'        and to_date(last_saved,'mm/dd/yyyy hh24:mi:ss')  >  (SYSDATE) - interval '10' minute        and opb_object_type.object_type_id=REP_VERSION_PROPS.OBJECT_TYPE
